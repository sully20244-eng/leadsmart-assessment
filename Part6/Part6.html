# Nightly Batch Scoring Job (Airflow DAG)
from datetime import datetime, timedelta
from airflow import DAG
from airflow.operators.python import PythonOperator

def batch_score_leads():
    # 1. Fetch all active leads (last 90 days, not converted)
    active_leads = get_active_leads_from_db(days=90)
    # 2. Load latest model from model registry
    model = load_model("leadsmart_champion", version="v3.2")
    # 3. Batch feature engineering
    features = compute_batch_features(active_leads)
    # 4. Parallel prediction
    scores = model.predict_proba(features)[:, 1]
    # 5. Update database
    update_lead_scores_in_db(active_leads.lead_id, scores)
    # 6. Generate analytics report
    generate_performance_report(scores)

# Schedule: Daily at 2 AM
dag = DAG(
    'batch_lead_scoring',
    schedule_interval='0 2 * * *', # Daily at 2 AM
    default_args={'retries': 3}

Model Performance Monitoring
# Monitoring pipeline for model drift
from evidently import ColumnMapping
from evidently.report import Report
from evidently.metrics import (
    DatasetDriftMetric,
    DatasetMissingValuesMetric,
    ColumnDriftMetric,
    ClassificationQualityMetric
)

def monitor_model_drift(reference_data, current_data):
    # Create monitoring report
    column_mapping = ColumnMapping(
        prediction='prediction',
        numerical_features=numerical_features,
        categorical_features=categorical_features
    )
    
    report = Report(metrics=[
        DatasetDriftMetric(),
        DatasetMissingValuesMetric(),
        ColumnDriftMetric(column_name='prediction'),
        ClassificationQualityMetric()
    ])
    
    report.run(
        reference_data=reference_data,
        current_data=current_data,
        column_mapping=column_mapping
    )
    
    if report.as_dict()['metrics'][0]['result']['dataset_drift']:
        trigger_retraining() # Alert and schedule retraining
